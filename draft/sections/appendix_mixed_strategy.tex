\section{Mixed-strategy indifference and the ODE}
\label{app:mixed-eqm}

\subsection{Incremental indifference}
\label{app:indifference}

On $[\underline{r},\bar{r}]$, equilibrium requires
$\hat{\Pi}(r)=\Pi(\underline{r}^{*})$.
The $\psi$ block in \eqref{eq:Pi-hat-simplified} does not depend on $r$; with
$\underline{r}=\underline{r}^{*}$ on path,
\begin{align}
  &\delta N (c_H - c_L)
    \Biggl[
      F\!\bigl(\beta^{-1}(r)\bigr)^{N}
      - F\!\bigl(\beta^{-1}(\underline{r}^{*})\bigr)^{N}
      - N \int_{\beta^{-1}(\underline{r}^{*})}^{\beta^{-1}(r)}
        F\!\left(
          \frac{
            (1 - \psi \rho(\beta(v))) c_H
            - \psi(1 - \rho(\beta(v))) c_L
          }{1 - \psi}
        \right)
        F(v)^{N-2}\, dF(v)
    \Biggr]
  \notag\\
  &\quad + N \Biggl\{
    (r - c_L) F(r)^{N-1} \bigl(1 - F(\beta^{-1}(r))\bigr)
    + \int_{\beta^{-1}(r)}^{1}
      \int_{r}^{v} (s - c_L)\, dF(s)^{N-1}
      \, dF(v)
  \Biggr\}
  \notag\\
  &= N \Biggl\{
    (\underline{r}^{*} - c_L) F(\underline{r}^{*})^{N-1}
      \bigl(1 - F(\beta^{-1}(\underline{r}^{*}))\bigr)
    + \int_{\beta^{-1}(\underline{r}^{*})}^{1}
      \int_{\underline{r}^{*}}^{v} (s - c_L)\, dF(s)^{N-1}
      \, dF(v)
  \Biggr\}.
  \label{eq:indifference}
\end{align}

\subsection{Derivative and the ODE}
\label{app:indifference-diff}

On the low branch, $\beta^{-1}(r)=h(r)$. Differentiating
\eqref{eq:indifference} in $r$ gives
\begin{align}
  0
  &= \delta N (c_H - c_L)
    \Biggl[
      F\!\bigl(h(r)\bigr)^{N-1}
        f\!\bigl(h(r)\bigr)\,h'(r)
      - F\!\left(
          \frac{
            (1 - \psi \rho(r)) c_H
            - \psi(1 - \rho(r)) c_L
          }{1 - \psi}
        \right)
        F\!\bigl(h(r)\bigr)^{N-2}
        f\!\bigl(h(r)\bigr)\,h'(r)
    \Biggr]
  \notag\\
  &\quad + F(r)^{N-1} \bigl(1 - F(h(r))\bigr)
  - (r - c_L) F(r)^{N-1}
    f\!\bigl(h(r)\bigr)\,h'(r)
  \notag\\
  &\quad - f\!\bigl(h(r)\bigr)\,h'(r)
    \int_{r}^{h(r)} (s - c_L)\, dF(s)^{N-1}.
  \label{eq:indifference-diff}
\end{align}
Collecting terms with $h'(r)$ in \eqref{eq:indifference-diff} and solving gives
\eqref{eq:ode-h}--\eqref{eq:B-bracket}.

\subsection{Initial denominator}
\label{app:B-FOC}

At the lower support, $\rho(\underline{r}^{*})=0$, so buyers keep the prior
belief $\psi$ after failure.  The first bidding branch in
\eqref{eq:beta-piecewise} therefore gives
\begin{equation*}
  \beta^{-1}(\underline{r}^{*})
  = \frac{\underline{r}^{*}-\delta\psi c_L}{1-\delta\psi}.
\end{equation*}
The stage-2 cutoff in \eqref{eq:B-bracket} also reduces to
$(c_H-\psi c_L)/(1-\psi)$.  To evaluate the ODE denominator at the initial
point, first use
\begin{equation*}
  \begin{aligned}
  &\int_{\underline{r}^{*}}^{\beta^{-1}(\underline{r}^{*})}
    (s-c_L)\,dF(s)^{N-1} \\
  &\quad =
    \bigl(\beta^{-1}(\underline{r}^{*})-c_L\bigr)
    F\!\bigl(\beta^{-1}(\underline{r}^{*})\bigr)^{N-1}
    - (\underline{r}^{*}-c_L)F(\underline{r}^{*})^{N-1}
    - \int_{\underline{r}^{*}}^{\beta^{-1}(\underline{r}^{*})}
      F(v)^{N-1}\,dv .
  \end{aligned}
\end{equation*}
Substituting this identity into \eqref{eq:B-bracket} and using
\eqref{eq:FOC-low-simplified} gives
\begin{align*}
  &\mathcal{B}\bigl(\underline{r}^{*},
    \beta^{-1}(\underline{r}^{*})\bigr)
  \\
  &= (1-\delta\psi)\,
    \frac{1 - F\!\bigl(\beta^{-1}(\underline{r}^{*})\bigr)}{
      f\!\bigl(\beta^{-1}(\underline{r}^{*})\bigr)
    }\,
    F(\underline{r}^{*})^{N-1}
  \\
  &\quad + \delta (c_H - c_L)
      F\!\left(\frac{c_H - \psi c_L}{1 - \psi}\right)
      F\!\bigl(\beta^{-1}(\underline{r}^{*})\bigr)^{N-2}.
\end{align*}
This is the initial denominator used in Claim~\ref{claim:ode-existence}.  It is
strictly positive whenever
$F(\beta^{-1}(\underline{r}^{*}))<1$.

\subsection{High reserves}
\label{app:high-r}

For $r>\beta(v^{*})$, $\beta^{-1}(r)=(r-\delta c_H)/(1-\delta)$ and
\eqref{eq:indifference-diff} becomes
\begin{align}
  0
  &= \delta N (c_H - c_L)
    \left(
      F\!\left(\frac{r - \delta c_H}{1 - \delta}\right)
      - F\!\left(
        \frac{
          (1 - \psi \rho(r)) c_H
          - \psi(1 - \rho(r)) c_L
        }{1 - \psi}
      \right)
    \right)
    F\!\left(\frac{r - \delta c_H}{1 - \delta}\right)^{N-2}
    f\!\left(\frac{r - \delta c_H}{1 - \delta}\right)
    \frac{1}{1 - \delta}
  \notag\\
  &\quad + F(r)^{N-1}
    \left(1 - F\!\left(\frac{r - \delta c_H}{1 - \delta}\right)\right)
  \notag\\
  &\quad - \frac{1}{1 - \delta}
    f\!\left(\frac{r - \delta c_H}{1 - \delta}\right)
    \Biggl[
      \left(\frac{r - \delta c_H}{1 - \delta} - c_L\right)
        F\!\left(\frac{r - \delta c_H}{1 - \delta}\right)^{N-1}
      - \int_{r}^{(r - \delta c_H)/(1 - \delta)} F(v)^{N-1}\, dv
    \Biggr],
  \label{eq:indifference-high-r}
\end{align}
which pins down $\rho(r)$ on the upper part of the support.
