\section{Preliminary Analysis}
\label{sec:preliminary}

\subsection{Second-stage belief and offer}
\label{subsec:second-stage}

If the chosen buyer has a belief $\tilde{\psi}$, he offers price $c_H$ if and
only if
\begin{equation}
  v - c_H \ge \tilde{\psi}(v - c_L)
  \quad\Longleftrightarrow\quad
  v \ge \frac{c_H - \tilde{\psi} c_L}{1 - \tilde{\psi}}.
  \label{eq:second-stage-threshold}
\end{equation}

\subsection{Equilibrium structure}
\label{subsec:eqm-structure}

Consider an equilibrium where the high-cost seller plays a pure strategy $r_H$
and the low-cost seller plays a mixed strategy denoted by
$\rho \in \Delta[\underline{r}, \bar{r}]$.

\begin{claim}
  \label{claim:rH-bar}
  $r_H > \bar{r}$.
\end{claim}
For any increasing bidding function, the optimal reserve price for a high-cost
seller is strictly above that of a low-cost seller.

\subsection{High-cost seller}
\label{subsec:high-cost-prelim}

When the bid exceeds $\bar{r}$ but the reserve is not met, bidders believe the
seller is high-cost and bid according to $\beta(v) = v - \delta(v - c_H)$.
Once a bidder has $\beta(v) \ge r_H$, the reserve is announced as met and
people bid their own values. The high-cost seller's profit $\Pi_H(r_H)$ is
\begin{equation}
  \Pi_H(r_H)
  = \int_{\beta^{-1}(r_H)}^{1}
    \int_{c_L}^{v}
      \bigl(\max\{r_H, s\} - c_H\bigr)
      \, d\!\left(\frac{F(s)}{F(v)}\right)^{N-1}
    \, dF(v)^{N}.
  \label{eq:Pi-H-integral}
\end{equation}

Simplifying,
\begin{align}
  \Pi_H(r_H)
  &= N \int_{\beta^{-1}(r_H)}^{1}
    \Bigl[
      (r_H - c_H) F(r_H)^{N-1}
      + \int_{r_H}^{v} (s - c_H)\, dF(s)^{N-1}
    \Bigr]
    \, dF(v)
  \notag\\
  &= N \Biggl[
    \begin{aligned}[t]
      &(r_H - c_H) F(r_H)^{N-1}
        \bigl(1 - F(\beta^{-1}(r_H))\bigr)
      + \int_{r_H}^{1} (s - c_H)\, dF(s)^{N-1} \\
      &- F(\beta^{-1}(r_H))
        \int_{r_H}^{\beta^{-1}(r_H)} (s - c_H)\, dF(s)^{N-1}
      - \int_{\beta^{-1}(r_H)}^{1}
        F(v)(v - c_H)\, dF(v)^{N-1}
    \end{aligned}
  \Biggr].
  \label{eq:Pi-H-expanded}
\end{align}

The first-order condition is
\begin{equation}
  r_H - c_H
  = (1 - \delta)
    \frac{
      1 - F\!\left(\dfrac{r_H - \delta c_H}{1 - \delta}\right)
    }{
      f\!\left(\dfrac{r_H - \delta c_H}{1 - \delta}\right)
    }
  - \frac{1}{F(r_H)^{N-1}}
    \int_{r_H}^{(r_H - \delta c_H)/(1 - \delta)}
      (v - c_H)\, dF(v)^{N-1}.
  \label{eq:FOC-high-cost}
\end{equation}

\subsection{Belief after a failed English auction}
\label{subsec:belief-failed}

If the English auction fails with final bid $b$, the buyer's belief is
\begin{equation}
  \tilde{\psi}
  = \frac{\psi\bigl(1 - \rho(b)\bigr)}{1 - \psi \rho(b)}.
  \label{eq:belief-update}
\end{equation}
