\section{Profit for \texorpdfstring{$r>\underline{r}$}{r above underline r}}
\label{app:profit-above}

Write $z\equiv\beta^{-1}(\underline{r})$ and $z(r)\equiv\beta^{-1}(r)$ as in
\eqref{eq:z-def}, $v_{\psi}$ from \eqref{eq:v-psi-def}, and
$v_{\rho}(v)$ as in \eqref{eq:Pi-true-fail}.
The stage-2 failure block in \eqref{eq:Pi-hat-simplified} is
\begin{align}
  &\delta N (c_H - c_L)\,
    \bigl(F(z) - F(v_{\psi})\bigr)\,
    F(z)^{N-1}
  \notag\\
  &\quad + \delta N (c_H - c_L)
    \int_{z}^{z(r)}
      \left(1 - \frac{F\bigl(v_{\rho}(v)\bigr)}{F(v)}\right)\, dF(v)^{N},
  \label{eq:Pi-hat-initial}
\end{align}
plus the stage-1 success terms in \eqref{eq:Pi-hat-simplified}.
The first line is \eqref{eq:profit-fail-closed} at $\underline{r}$.
The integral over $(z,z(r)]$ collects updated-belief payoffs as in the
$(z,z^{*}]$ terms of \eqref{eq:Pi-true-fail} with $\underline{r}^{*}$ replaced
by $r$.
Since
$\int (1-F_{\rho}/F)\,dF^{N}
= N\int (F-F_{\rho})F^{N-2}\,dF$ with $F_{\rho}\equiv F\bigl(v_{\rho}(v)\bigr)$,
\begin{equation*}
  \int_{z}^{z(r)}
    \left(1 - \frac{F_{\rho}}{F}\right)\, dF^{N}
  = F\bigl(z(r)\bigr)^{N}
    - F(z)^{N}
    - N \int_{z}^{z(r)}
      F_{\rho}\,F^{N-2}\, dF,
\end{equation*}
which yields the middle terms in \eqref{eq:Pi-hat-simplified}.
